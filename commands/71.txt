BEGIN
    UPDATE employees
    SET salary = salary + 1000
    WHERE emp_id = 101;

    IF SQL%FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Employee updated successfully.');
        DBMS_OUTPUT.PUT_LINE('Rows affected: ' || SQL%ROWCOUNT);
    ELSE
        DBMS_OUTPUT.PUT_LINE('No employee found with emp_id 101.');
    END IF;
END;
